/*TABLA DE PRODUCTOS */
CREATE TABLE productos (
    	id VARCHAR(15) PRIMARY KEY,
    	nombre VARCHAR(150) NOT NULL,
    	descripcion TEXT,
    	categoria VARCHAR(50) NOT NULL, 
	tipo VARCHAR(50) NOT NULL, 
    	precio DECIMAL(10,2) NOT NULL,
	imagen BYTEA,
	createAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	updateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);



/* TABLA DE CLIENTES */
CREATE TABLE clientes (
    id VARCHAR(10) PRIMARY KEY DEFAULT gen_random_uuid(),
    nombre VARCHAR(150) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    password TEXT NOT NULL,
    telefono VARCHAR(20),
    usuario_id varchar(20),
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_usuario
    FOREIGN KEY (usuario_id)
    REFERENCES usuarios(id)
);



/*  TABLA DE ALIADOS */
CREATE TABLE aliados (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    nombre TEXT NOT NULL,
    img BYTEA,
    createdAt TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updatedAt TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);



/* TABLA DE LA COMPRA DE CLIENTE */
CREATE TABLE comprasCliente (
    id VARCHAR(35) PRIMARY KEY,
    fecha DATE NOT NULL,
    total DOUBLE PRECISION NOT NULL,
    status TEXT NOT NULL,
    id_cliente VARCHAR(15) NOT NULL,
    email VARCHAR(255),
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_cliente
    FOREIGN KEY (id_cliente)
    REFERENCES clientes(id)
);



/* TABLA DE DETALLE DE COMPRA DE LOS CLIENTES */
CREATE TABLE detalleCliente (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    cantidad INT NOT NULL, 
    id_compra VARCHAR(35) NOT NULL,
    id_producto VARCHAR(15) NOT NULL,
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_compra
    FOREIGN KEY (id_compra)
    REFERENCES comprasCliente(id),

    CONSTRAINT fk_producto
    FOREIGN KEY (id_producto)
    REFERENCES productos(id)
);




/* Tabla de las compras a los proveedores */
CREATE TABLE comprasProveedores (
    id VARCHAR(35) PRIMARY KEY,
    fecha DATE NOT NULL,
    total DOUBLE PRECISION NOT NULL,
    status TEXT NOT NULL,
    id_aliado UUID NOT NULL,
    id_usuario VARCHAR(10) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_aliado
    FOREIGN KEY (id_aliado)
    REFERENCES aliados(id),

    CONSTRAINT fk_usuario
    FOREIGN KEY (id_usuario)
    REFERENCES usuarios(id)
);


/* TABLA DE DETALLE DE COMPRAS DE PROVEEDORES */
CREATE TABLE detalleProveedores (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    cantidad INT NOT NULL, 
    id_compra_proveedor VARCHAR(35) NOT NULL,
    id_producto VARCHAR(15) NOT NULL,
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_compra
    FOREIGN KEY (id_compra)
    REFERENCES comprasCliente(id),

    CONSTRAINT fk_producto
    FOREIGN KEY (id_producto)
    REFERENCES productos(id)
);



/* CREACIÓN DE LA TABLA DE USUARIOS */
CREATE TABLE usuarios (
    id VARCHAR(10) PRIMARY KEY DEFAULT gen_random_uuid(),
    nombre VARCHAR(150) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    password TEXT NOT NULL,
    telefono VARCHAR(20),
    id_rol int,
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_rol
    FOREIGN KEY (id_rol)
    REFERENCES roles(id)
);



/* TABLA DE TALLAS */
CREATE TABLE tallas (
    id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    descripcion VARCHAR(5),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);



/* TABLA DE STOCK */
CREATE TABLE stock (
    id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    cantidad VARCHAR(5),
    id_producto VARCHAR(15),
    id_talla int, 
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
	
    FOREIGN KEY (id_producto) REFERENCES productos(id),
    FOREIGN KEY (id_talla) REFERENCES tallas(id)
);


/* TABLA DE ROLES */
CREATE TABLE roles (
    id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    descripcion VARCHAR(100),
    status VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
);
